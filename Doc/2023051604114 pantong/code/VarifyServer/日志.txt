// 1. 引入路径工具（找文件用的，比如找message.proto文件）
const path =require('path')

// 2. 引入gRPC工具（用来启动服务，让别人能找到咱们）
const grpc=require('@grpc/grpc-js')

// 3. 引入proto加载工具（解析服务规则的，比如别人该怎么给咱们发请求）
const protoLoader =require('@grpc/proto-loader')

// 4. 找到message.proto文件（这个文件定义了“别人该怎么跟咱们沟通”，比如要传邮箱过来）
const PROTO_PATH=path.join(__dirname,'message.proto')

// 5. 解析proto文件，变成代码能懂的规则
const packageDefinition =protoLoader.loadSync(PROTO_PATH,{keepCase:true,longs:String,enum:String,defaults:true,oneofs:true})
const protoDescriptor=grpc.loadPackageDefinition(packageDefinition)
const message_proto=protoDescriptor.message
